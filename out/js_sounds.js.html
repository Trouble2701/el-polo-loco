/**
 * This function start the Startsound and replayed this sound after end
 */
function startSoundPlay() {
    startSound.play();
    setInterval(() => {
        if (time > 61 && pepeDead == 1) {
            startSound.pause();
            startSound.play();
            time = 0;
        }
        time++;
    }, 1000);
}
